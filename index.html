def merge_sort(list):
  if (len(list) == 1):
    return list

  half = int(len(list)/2)

  left_half = list[:half]
  right_half = list[half:]

  left = merge_sort(left_half)
  right = merge_sort(right_half)
  sorted_list = []

  while (len(left) != 0 or len(right) != 0):
    if len(right) == 0:
      sorted_list.append(left[0])
      del left[0]

    elif len(left) != 0 and left[0] <= right[0]:
      sorted_list.append(left[0])
      del left[0]

    else:
      sorted_list.append(right[0])
      del right[0]

  return sorted_list
